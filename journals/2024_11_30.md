- #EE 使用三个GPIO和三八译码器去驱动七段数码管
	- ```C
	  // NUM -> bits
	  const uint16_t NUMS[] = { 
	    63,
	    6,
	    91,
	    79,
	    102,
	    109,
	    125,
	    7,
	    127,
	    111
	  };
	  
	  void setup() {
	    pinMode(10, OUTPUT);
	    pinMode(11, OUTPUT);
	    pinMode(12, OUTPUT);
	  }
	  
	  void printNum(int num) {
	    for (int i = 0; i <= 6; i++) {
	      if ((num >> i) % 2 == 1) {
	        digitalWrite(10, i % 2 == 1);
	        digitalWrite(11, (i >> 1) % 2 == 1);
	        digitalWrite(12, (i >> 2) % 2 == 1);
	        delay(1);
	      }
	    }
	  }
	  
	  uint32_t lastMill = -1;
	  int lastNum = 0;
	  void loop() {
	    printNum(NUMS[lastNum]);
	    uint32_t nowMill = millis();
	    if (nowMill - lastMill >= 500) {
	      lastNum = (lastNum + 1) % 10;
	      lastMill = nowMill;
	    }
	  }
	  ```
